// Telegram channel messages
// url = https://tpy.onrender.com/tele-messages?user=&only_remove=
import requests
user=input('[?] Enter The user : ') #Like vv1ck
only_remove=input('[?] You want to see only deleted messages or all messages : ') #You must type "true" if you want to show only deleted messages, and when you want to display all messages, type anything that does not matter  
respone=requests.get(f'https://tpy.onrender.com/tele-messages?user={user}&only_remove={only_remove}').json()
print(respone)
        
// Respone should Be : 
{"Telegram":"@TweakPY & @vv1ck","response":{"messages":[{"deleted":"25 Feb, 00:29","msg":"Hi","posted":"25 Feb, 00:28"}]}}
